```c
#ifdef DEBUG

#else
#define PRINT_DEBUG(n)
#endif
```

```c
#define DEBUG
#include "debug.h"
int main(void)
{

#ifdef DEBUG

#else
    printf("Output if DEBUG is not defined:\n");
#endif

    PRINT_DEBUG(i);
    PRINT_DEBUG(j);
    PRINT_DEBUG(k);
    PRINT_DEBUG(i + j);
    PRINT_DEBUG(2 * i + j - k);

    return 0;
}
```

(c) Explain why the output is different in parts (a) and (b).
